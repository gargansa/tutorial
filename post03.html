Recursion

Repetition with an exit strategy

Posted by Robert Crum last updated October 21, 2019

What is recursion

A way of simplifying a problem into smaller problems. Divide and Conquer.

Starting Simple

This function contains every piece you need for a recursive function.

function myfunction(i){

console.log(i)

if (i>0){

myfunction(i-1)

}

}

myfunction(3)

Making something useful

Alphabetized list words

function sortFunction(arr) {

if (Array.isArray(arr)) {

for (var piece = 0; piece < arr.length; piece++) {

arr[piece] = sortFunction(arr[piece]);

}

arr = arr.sort();

}

return arr;

}

var myArray = ["bear","jellyfish","cat","waterbuffalo","eagle","anteater","fish"];

sortFunction(myArray);

console.log(myArray);

var multiArray = [["car", "truck", "boat"],["birch", "oak", "pine"],["watermelon"],["french", "spanish"],["one", "two", "three", "four"]];

sortFunction(multiArray);

console.log(multiArray);

var nameArray = [["name","John Smith"],["Phone","555-555-555"],["address",["city","Bozeman"]]]

sortFunction(nameArray);

console.log(nameArray);
